        // En esta parte se declaran las bibliotecas que se van autilizar

//Aqui se declaran las funciones a implementar
void set();//Funcion que envia el valor final
int potencia(int a);//Funcion que calcula la potencia del numero elevado a ese mismo numero

/*Aqui se declara la variable global final que guarda
el valor final de la serie*/
int final;

//Aqui se define el programa principal 
int main(){
       float suma_total=0;//Declararcion y asignacion del acumulador de la suma total de la operacion
       int n;//Esta variable funge como una copia del valor final
       set();//Se llama a la funcion de ingreso de datos
       n=final;//Se le asigna el valor de final a n para igualarlas
       float cociente;//Esta variable guardara el valor del coicente de ii/i!       
       
       /*Se inicializa el ciclo do-while*/
       
       do{                    
       float pot=potencia(n);//Se declara pot y se le asigna el valor que regresa la funcion potencia 
       float fac=factorial(n);//Se declara fac y se le asigna el valor que regresa la funcion factorial        
       cociente=pot/fac;//Se le asigna a la variable cociente el cociente de pot/fac       
       
       
       /*
       
        Lineas de apoyo para saber si el programa funciona correctamente
       //printf("\nEl valor de potencia es %.2f",pot);

       */
       
       suma_total=suma_total+cociente;//Se va acumulando el valor de la suma de los cocientes
       
       //printf("\nEl valor de n es: %d",n);
       
       n=n-1;//Se decrementa el valor  de la variable n que corresponde al valor de final para que el ciclo regrese
       
       //printf("\nEl valor de n es: %d",n);
         }while(n>=1);//Se da la condicion para que el ciclo se siga ejecutando
       printf("\nEl valor total es: %.2f\n",suma_total);//Se imprime el valor del total de la sumatoria 
       return 0;//Indica terminacion exitosa del programa
      }
